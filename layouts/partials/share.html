page overrides site config */}} {{- $enabled := cond (isset .Params "share_enabled") $pageEnabled $globalEnabled -}} {{- $services := cond (isset .Params "share_services") $pageServices $globalServices -}} {{- if $enabled -}} {{- $shareData := .Site.Data.shareservices | default dict -}} {{- $pageURL := .Permalink -}} {{- $pageTitle := .Title -}} {{/* Build enticing share text based on page type */}} {{- $shareText := .Title -}} {{- if and .Params.hacker .Params.conference -}} {{/* Talk / entry page — mention speakers & conference */}} {{- $hackerNames := slice -}} {{- range .Params.hacker -}} {{- $hackerNames = $hackerNames | append (. | humanize | title) -}} {{- end -}} {{- $speakers := delimit $hackerNames " & " -}} {{- $conf := .Params.conference | humanize | title -}} {{- $shareText = printf "\"%s\" by %s at %s — worth your time if you're into security." .Title $speakers $conf -}} {{- else if eq .Section "hacker" -}} {{/* Hacker profile page */}} {{- $shareText = printf "Check out %s's security research — talks, tools & contributions, all in one place." .Title -}} {{- else if eq .Section "book" -}} {{/* Book page */}} {{- $shareText = printf "\"%s\" — solid read for anyone in security." .Title -}} {{- else if eq .Section "conference" -}} {{/* Conference page */}} {{- $shareText = printf "Some great security talks from %s — worth diving into." .Title -}} {{- else if eq .Section "tool" -}} {{/* Tool page */}} {{- $shareText = printf "%s — security tool worth checking out." .Title -}} {{- end -}}
