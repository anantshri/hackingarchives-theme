{{/* Share Partial - Pure URL-based social sharing (No JavaScript) Configuration (in config.toml under [params.share]): enabled = true # Enable/disable share links globally services = ["twitter", "linkedin", "email"] # Which services to show Per-page override (in frontmatter): share_enabled: false # Disable on this page share_services: ["twitter", "email"] # Override services for this page Usage in templates: {{ partial "share.html" . }} Renders as a share-box (positioned top-right by CSS) with share icon + service buttons. */}} {{- $shareConfig := .Site.Params.share | default dict -}} {{- $globalEnabled := $shareConfig.enabled | default false -}} {{- $globalServices := $shareConfig.services | default (slice "twitter" "linkedin" "email") -}} {{/* Per-page overrides */}} {{- $pageEnabled := .Params.share_enabled -}} {{- $pageServices := .Params.share_services -}} {{/* Determine final settings - page overrides site config */}} {{- $enabled := cond (isset .Params "share_enabled") $pageEnabled $globalEnabled -}} {{- $services := cond (isset .Params "share_services") $pageServices $globalServices -}} {{- if $enabled -}} {{- $shareData := .Site.Data.shareservices | default dict -}} {{- $pageURL := .Permalink -}} {{- $pageTitle := .Title -}}
